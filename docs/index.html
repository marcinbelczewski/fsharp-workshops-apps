let sequenceOpts (optionals: 'a option list) : 'a list option =
    let rec sequenceOpts' acc optionals =
        match optionals, acc with
        | [],_ ->
            Option.map List.rev acc
        | Some h :: t, Some acc ->
            sequenceOpts' (Some (h :: acc)) t
        | _ ->
            None

    sequenceOpts' (Some []) optionals
